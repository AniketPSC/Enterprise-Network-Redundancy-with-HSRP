! L3SW-2 (Standby L3 switch)
hostname L3SW-2
no ip domain-lookup
ip routing

! VLANs
vlan 10
vlan 20
vlan 30
exit

! Trunk to access
interface Gi0/1
 switchport mode trunk
 switchport trunk allowed vlan 10,20,30
 no shutdown
exit

! Routed link to R-HQ (secondary link)
interface Gi0/2
 no switchport
 ip address 10.0.0.6 255.255.255.252
 no shutdown
exit

! SVIs + HSRP (standby)
interface Vlan10
 ip address 10.0.10.3 255.255.255.0
 standby 10 ip 10.0.10.1
 standby 10 priority 100
 no shutdown
exit

interface Vlan20
 ip address 10.0.20.3 255.255.255.0
 standby 20 ip 10.0.20.1
 standby 20 priority 100
 no shutdown
exit

interface Vlan30
 ip address 10.0.30.3 255.255.255.0
 standby 30 ip 10.0.30.1
 standby 30 priority 100
 no shutdown
exit

router ospf 1
 router-id 2.2.2.2
 network 10.0.0.4 0.0.0.3 area 0
 network 10.0.10.0 0.0.0.255 area 0
 network 10.0.20.0 0.0.0.255 area 0
 network 10.0.30.0 0.0.0.255 area 0
exit

! Make sure policy mirrored on standby for consistency
ip access-list extended HR_TO_IT_BLOCK
 deny ip 10.0.10.0 0.0.0.255 10.0.30.0 0.0.0.255
 permit ip any any
exit
interface Vlan10
 ip access-group HR_TO_IT_BLOCK in
exit

write memory
